1   package es.ull.flightspassengers;
2   
3   import static org.junit.jupiter.api.Assertions.assertAll;
4   import static org.junit.jupiter.api.Assertions.assertEquals;
5   
6   import org.junit.jupiter.api.BeforeEach;
7   import org.junit.jupiter.api.DisplayName;
8   import org.junit.jupiter.api.Nested;
9   import org.junit.jupiter.api.Test;
10  
11  import es.ull.flights.Flight;
12  import es.ull.passengers.Passenger;
13  
14  public class FlightPassengersTest {
15  
16  	@DisplayName("Tests intercambio correcto pasajeros")
17  	@Nested
18  	class InterchangeFlightsPassengers {
19  
20  	    private Flight vueloFuerteventura;
21  	    private Flight vueloLanzarote;
22  	    private Passenger jaime;
23  
24  	    @BeforeEach
25  	    void setUp() {
26  	    	vueloFuerteventura = new Flight("FV001", 100);
27  	    	vueloLanzarote = new Flight("LZ001", 82);
28  	    	jaime = new Passenger("id-PS01-2","Jaime", "ES");
29  	    }
30  
31  	        @Test
32  	        @DisplayName("Intercambio correcto entre vuelos a través de joinFlight")
33  	        void testMethodInterchangePassenger() {
34  
35  	        	jaime.joinFlight(vueloLanzarote);
36  	        	jaime.joinFlight(vueloFuerteventura);
37  
38  	            assertAll("Verifica todas las condiciones un intercambio a través de joinFlight",
39  	                    () -> assertEquals("FV001", jaime.getFlight().getFlightNumber()),
40  	                    () -> assertEquals(1, vueloFuerteventura.getNumberOfPassengers()),
41  	                    () -> assertEquals(0, vueloLanzarote.getNumberOfPassengers())
42  	            );
43  	        }
44  
45  	        @Test
46  	        @DisplayName("Intercambio manual entre vuelos a través de setFlight and removePassenger")
47  	        void testManualInterchangeMethod() {
48  
49  	        	jaime.joinFlight(vueloFuerteventura);
50  	        	jaime.setFlight(vueloLanzarote);
51  
52  	        	assertAll("Verifica todas las condiciones un intercambio a través de removePassenger y setFlight",
53  	        	() -> assertEquals("LZ001", jaime.getFlight().getFlightNumber()),
54  	        	() -> assertEquals(1, vueloFuerteventura.getNumberOfPassengers()),
55  	        	() -> assertEquals(0, vueloLanzarote.getNumberOfPassengers()),
56  	        	() -> {
57  	        		vueloLanzarote.addPassenger(jaime);
58  	        		assertEquals(1, vueloLanzarote.getNumberOfPassengers());
59  	        	},
60  	        	() -> assertEquals(1, vueloFuerteventura.getNumberOfPassengers()),
61  	        	() -> {
62  	        		vueloFuerteventura.removePassenger(jaime);
63  	        		assertEquals(0, vueloFuerteventura.getNumberOfPassengers());
64  	        	});
65  	        }
66  	    }
67  }
